module WindowAlert =
    open WebSharper
    open WebSharper.Html.Client
    open WebSharper.JavaScript

    type Control() =
        inherit Web.Control()

        [<JavaScript>]
        override __.Body =
            Button [Class "btn btn-primary btn-lg"]
            -- Text "Click me"
            |>! OnClick (fun _ _ -> JS.Window.Alert "This is an alert dialog.")
            :> _
