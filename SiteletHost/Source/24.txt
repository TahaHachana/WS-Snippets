module WindowAlert =
    open IntelliFactory.WebSharper
    open IntelliFactory.WebSharper.Html
    open IntelliFactory.WebSharper.Html5

    type Control() =
        inherit Web.Control()

        [<JavaScript>]
        override __.Body =
            Button [Attr.Class "btn btn-primary btn-lg"]
            -- Text "Click me"
            |>! OnClick (fun _ _ -> Window.Self.Alert "This is an alert dialog.")
            :> _
