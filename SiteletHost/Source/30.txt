module KendoUICalendar =
    open IntelliFactory.WebSharper
    open IntelliFactory.WebSharper.Html
    open IntelliFactory.WebSharper.KendoUI

    type Control() =
        inherit Web.Control()

        [<JavaScript>]
        override __.Body =
            Div []
            |>! OnAfterRender (fun elt ->
                kendo.ui.Calendar(elt.Body)
                |> ignore)
            :> _
