module HTML5Audio =
    open WebSharper
    open WebSharper.Html.Client
    open WebSharper.JavaScript

    type Control() =
        inherit Web.Control()

        [<JavaScript>]
        override __.Body =
            let elt = Audio []
            let audio = As<HTMLAudioElement> elt.Dom
            audio.Src <- "/AlFatiha.mp3"
            audio.Controls <- true
            elt :> _
